00226 static uint8_t virtual_keyboard_translate_to_hid(int c)
00227 {
00228   if (c <= 0)
00229   {
00230     return 0;
00231   }
00232   else if ('a' <= c && c <= 'z')
00233   {
00234     return (c - 'a' + USB_HID_KEY_a_A);
00235   }
00236   else if ('A' <= c && c <= 'Z')
00237   {
00238     return (c - 'A' + USB_HID_KEY_a_A) | LEFT_SHIFT_BIT;
00239   }
00240   else if ('1' <= c && c <= '9')
00241   {
00242     return (c - '1' + USB_HID_KEY_1);
00243   }
00244   else if ('0' == c)
00245   {
00246     return (USB_HID_KEY_0);
00247   }
00248   else if ('.' == c)
00249   {
00250     return (USB_HID_KEY_DOT);
00251   }
00252   else if (',' == c)
00253   {
00254     return (USB_HID_KEY_COMMA);
00255   }
00256   else if ('\r' == c)
00257   {
00258     return (USB_HID_KEY_RETURN);
00259   }
00260   else if (' ' == c || '\t' == c)
00261   {
00262     return (USB_HID_KEY_SPACEBAR);
00263   }
00264   else
00265   {
00266     return (USB_HID_KEY_SPACEBAR);

00299 static void hardwareInit(void)
00300 {
00301   uchar i, j;
00302   PORTD = 0xfa;
00303   /* 1111 1010 bin: activate pull-ups except on USB lines */
00304   DDRD = 0x07;
00305   /* 0000 0111 bin: all pins input except USB (-> USB reset) */
00306   j = 0;
00307   while (--j)
00308   { /* USB Reset by device only required on Watchdog Reset */
00309     i = 0;
00310     while (--i)
00311       ; /* delay >10ms for USB reset */
00312   }
00313   DDRD = 0x02;
00314   /* 0000 0010 bin: remove USB reset condition */
00315   /* configure timer 0 for a rate of 12M/(1024 * 256) = 45.78 Hz (~22ms) */
00316   TCCR0 = 5;
00317   /* timer 0 prescaler: 1024 */
00318 }

00465 uchar usbFunctionSetup(uchar data[8])
00466 {
00467   usbRequest_t* rq = (void *)data;
00468   usbMsgPtr = reportBuffer;
00469   if((rq->bmRequestType & USBRQ_TYPE_MASK) == USBRQ_TYPE_CLASS)
00470   {
00471     /* class request type */
00472     if(rq->bRequest == USBRQ_HID_GET_REPORT)
00473     {
00474       /* wValue: ReportType (highbyte), ReportID (lowbyte) */
00475       /* we only have one report type, so don't look at wValue */
00476       buildReport(rq->wValue.bytes[0],ir_key_pressed,vtLastKey);
00477       return sizeof(reportBuffer);
00478     }
00479     else if(rq->bRequest == USBRQ_HID_GET_IDLE)
00480     {
00481       usbMsgPtr = &idleRate;
00482       return 1;
00483     }
00484     else if(rq->bRequest == USBRQ_HID_SET_IDLE)
00485     {
00486       idleRate = rq->wValue.bytes[1];
00487     }
00488   }
00489   else
00490   {
00491     /* no vendor specific requests implemented */
00492   }
00493   return 0;
